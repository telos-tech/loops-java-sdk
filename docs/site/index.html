Loops Java SDK

A modern, type-safe Java client for the Loops.so API with full async support

Get Started API Reference

Quick Start

// Initialize the client
LoopsClient client = LoopsClient.builder()
        .apiKey("your-api-key")
        .build();

// Create a contact
ContactResponse response = client.contacts().create(
    ContactCreateRequest.builder()
        .email("user@example.com")
        .firstName("John")
        .lastName("Doe")
        .build()
);

// Send an event
client.events().send(
    EventRequest.builder()
        .email("user@example.com")
        .eventName("signed_up")
        .build()
);

// Send a transactional email
client.transactional().send(
    TransactionalRequest.builder()
        .transactionalId("welcome_email")
        .email("user@example.com")
        .build()
);

Common Use Cases

Create and Update Contacts

// Create a new contact
ContactResponse response = client.contacts().create(
    ContactCreateRequest.builder()
        .email("user@example.com")
        .firstName("Jane")
        .customProperty("plan", "premium")
        .build()
);

// Update an existing contact
client.contacts().update(
    ContactUpdateRequest.builder()
        .email("user@example.com")
        .customProperty("lastLogin", LocalDateTime.now().toString())
        .build()
);

Send Events with Data

// Send an event with custom properties
client.events().send(
    EventRequest.builder()
        .email("user@example.com")
        .eventName("purchase_completed")
        .eventProperty("amount", 99.99)
        .eventProperty("product", "Premium Plan")
        .build()
);

Asynchronous Operations

// Run multiple operations in parallel
CompletableFuture<ContactResponse> createContact =
    client.contacts().createAsync(contactRequest);

CompletableFuture<EventResponse> sendEvent =
    client.events().sendAsync(eventRequest);

// Wait for both to complete
CompletableFuture.allOf(createContact, sendEvent)
    .thenRun(() -> System.out.println("All operations completed"))
    .join();
